```python
import streamlit as st
import random
import time

st.title("🌟 명언 생성기")

# 명언 데이터 (카테고리별 5개씩 총 20개 이상)
quotes = {
    "성공": [
        "성공은 작은 노력이 쌓여 이루어진다.",
        "실패는 성공의 어머니다.",
        "노력 없는 성공은 없다.",
        "포기하지 않는 것이 최고의 전략이다.",
        "꾸준함이 가장 강력한 힘이다."
    ],
    "도전": [
        "새로운 도전을 두려워하지 말자.",
        "꿈을 꾸는 것만으로는 부족하다. 행동하라!",
        "성공하는 사람은 실패를 두려워하지 않는다.",
        "도전 없이는 발전도 없다.",
        "모든 위대한 성취는 작은 도전에서 시작된다."
    ],
    "행복": [
        "행복은 순간이 아니라 과정이다.",
        "웃으면 복이 온다.",
        "오늘 하루를 소중히 여겨라.",
        "행복은 멀리 있는 것이 아니라 지금 이 순간에 있다.",
        "감사하는 마음이 행복을 만든다."
    ],
    "우정": [
        "진정한 친구는 어려울 때 빛난다.",
        "우정은 작은 배려에서 시작된다.",
        "좋은 친구는 인생 최고의 보물이다.",
        "친구와 함께하면 어떤 일도 즐겁다.",
        "우리는 서로에게 좋은 영향을 줄 수 있는 친구가 되어야 한다."
    ]
}

# 1️⃣ 명언 카테고리 선택
category = st.selectbox("📌 명언 카테고리를 선택하세요:", list(quotes.keys()))

# 2️⃣ 명언 로딩 애니메이션
if st.button("✨ 명언 보기"):
    with st.spinner("🔍 명언을 찾는 중..."):
        time.sleep(2)  # 2초 동안 로딩 애니메이션 표시

    # 명언 출력
    selected_quote = random.choice(quotes[category])
    st.success(f"📢 {selected_quote}")
```
